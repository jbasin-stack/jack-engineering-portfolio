Wire keyboard shortcuts (Ctrl+S, Escape) to real admin panel state and gate all admin imports from production builds. This closes INT-01, INT-02, INT-03 from the v1.1 milestone audit and hardens INFRA-01.

```typescript
import { lazy, Suspense, useState, useCallback, useEffect } from 'react';
// ... portfolio imports (NO admin imports) ...

const AdminShell = import.meta.env.DEV
  ? lazy(() => import('./admin/AdminShell'))
  : null;

function App() {
  // DEV-only admin state
  const [adminOpen, setAdminOpen] = useState(
    () => import.meta.env.DEV && new URLSearchParams(window.location.search).has('admin')
  );

  const toggleAdmin = useCallback(() => {
    setAdminOpen((prev) => {
      const next = !prev;
      const url = new URL(window.location.href);
      if (next) url.searchParams.set('admin', '');
      else url.searchParams.delete('admin');
      window.history.replaceState({}, '', url.toString());
      return next;
    });
  }, []);

  const closeAdmin = useCallback(() => {
    setAdminOpen(false);
    const url = new URL(window.location.href);
    url.searchParams.delete('admin');
    window.history.replaceState({}, '', url.toString());
  }, []);

  // DEV-only: Ctrl+Shift+A toggle shortcut (inline, not imported)
  useEffect(() => {
    if (!import.meta.env.DEV) return;
    const handler = (e: KeyboardEvent) => {
      if (e.ctrlKey && e.shiftKey && e.key === 'A') {
        e.preventDefault();
        toggleAdmin();
      }
    };
    window.addEventListener('keydown', handler);
    return () => window.removeEventListener('keydown', handler);
  }, [toggleAdmin]);

  return (
    <MotionConfig reducedMotion="user">
      {AdminShell && adminOpen && (
        <Suspense fallback={null}>
          <AdminShell onClose={closeAdmin} />
        </Suspense>
      )}
      <SmoothScroll>
        {/* ...portfolio sections unchanged... */}
      </SmoothScroll>
    </MotionConfig>
  );
}
```

- Zero static imports from `src/admin/` — the only admin reference is the dynamic `import()` inside `lazy()` behind `import.meta.env.DEV`.
- `adminOpen` state initializer uses `import.meta.env.DEV &&` so it's always `false` in production (and Vite eliminates the dead branch).
- `toggleAdmin` and `closeAdmin` become dead code in production (only referenced by DEV-gated code) and are tree-shaken.
- The `useEffect` for Ctrl+Shift+A has an early return on `!import.meta.env.DEV` so the listener is never registered in production.

```typescript
import { useKeyboardShortcuts } from './useKeyboardShortcuts';

export default function AdminShell({ onClose }: AdminShellProps) {
  const { isDirty, setDirty, ... } = useAdminPanel();
  const saveRef = useRef<(() => Promise<boolean>) | null>(null);
  const savingRef = useRef(false);

  const handleSave = async () => {
    if (saveRef.current) {
      await saveRef.current();
    }
  };

  // Save-in-progress guard for keyboard shortcut
  const guardedSave = useCallback(async () => {
    if (savingRef.current || !isDirty) return; // Already saving or nothing to save
    savingRef.current = true;
    try {
      await handleSave();
    } finally {
      savingRef.current = false;
    }
  }, [isDirty]); // handleSave is stable (depends only on saveRef)

  const handleClose = () => {
    if (isDirty) {
      const confirmed = window.confirm('You have unsaved changes. Close anyway?');
      if (!confirmed) return;
    }
    onClose();
  };

  // Keyboard shortcuts: Ctrl+S and Escape handled here with real state
  // isOpen is always true (AdminShell only renders when panel is open)
  // onToggle is no-op (Ctrl+Shift+A toggle lives in App.tsx)
  useKeyboardShortcuts(true, () => {}, guardedSave, handleClose, isDirty);

  // ... rest of component unchanged ...
};
```

Key points:
- `useKeyboardShortcuts` is called with `isOpen: true` (AdminShell only mounts when open), `onToggle: () => {}` (toggle lives in App.tsx), `guardedSave` (wraps handleSave with save-in-progress guard and isDirty check), `handleClose` (existing dirty confirmation logic), and `isDirty` (real value from useAdminPanel).
- `guardedSave` checks `savingRef.current` to prevent concurrent saves from rapid Ctrl+S.
- `guardedSave` checks `!isDirty` for the "silent no-op when no dirty changes" requirement.
- No changes to `useKeyboardShortcuts.ts` itself — the hook works correctly, we're just changing WHERE it's called and WHAT it receives.

**Do NOT modify `src/admin/useKeyboardShortcuts.ts`** — the hook API is correct as-is. Per research recommendation, keep it as-is and change the call site.
